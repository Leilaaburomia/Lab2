tree_growth <- tree_data %>%
  group_by(treeID) %>%
  arrange(year) %>%
  mutate(
    annual_DBH_growth = DBH_cm - lag(DBH_cm),

  ) %>%
  ungroup()

tree_growth_10yr <- tree_growth %>%
  group_by(treeID) %>%
  filter(n() >= 10) %>%
